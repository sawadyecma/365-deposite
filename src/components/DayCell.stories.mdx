import { useState } from "react";
import { Canvas, Meta, Story } from "@storybook/addon-docs";
import { DayCell } from "./DayCell";

<Meta title="MDX/DayCell" component={DayCell} />

export const Static = (args) => <DayCell {...args} />;
// もう少し綺麗に書けそう
export const Dynamic = (args) => {
  const [pressed, setPressed] = useState(false);
  return (
    <DayCell
      day={{ ...args.day, pressed }}
      onDayClick={(day) => {
        setPressed(!pressed);
      }}
    />
  );
};

<Canvas>
  <Story
    name="Pressed"
    args={{ day: { value: 1, pressed: true, pressedColor: "red" } }}
  >
    {Static.bind({})}
  </Story>
</Canvas>

<Canvas>
  <Story
    name="NotPressed"
    args={{ day: { value: 1, pressed: false, pressedColor: "red" } }}
  >
    {Static.bind({})}
  </Story>
</Canvas>

<Canvas>
  <Story
    name="Pressable"
    args={{
      day: { value: 1, pressed: pressed, pressedColor: "red" },
    }}
  >
    {Dynamic.bind({})}
  </Story>
</Canvas>
